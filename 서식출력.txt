# 서식 출력

# -> 문자열을 출력할 때, 서식문자와 함께 사용하는 것

# -> 서식문자 : %d %f %s

# -> %d 정수    %자릿수d (줄맞추려고)

# -> %f 실수    %자릿수.소수점자릿수(몇자리까지 할건지)f

# -> %s 문자열  %자릿수s

print("정수:%d, 실수:%f, 문자열:%s" %(5, 7.7, "hello"))

print("하나:%s, 둘:%s, 셋:%s" %(5, 7.7, "hello"))

print("정수:%10d, 실수:%10.1f, 문자열:%10s" %(5, 7.7, "hello"))

print("%c %c" %('A', 'B'))  # %c: 문자 하나만 쓸때

print("----------------------------")
